     ?strain a bacdive:Strain;

Question 4)  How are risk_assessment/pathogen human and met_antibiotica/ab_resistant (1) related. Possibly connected to culture temp (test_type=optimum) or strains (genus=Staphylococcus).


PREFIX owl: <http://www.w3.org/2002/07/owl#> 
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#> 
PREFIX xsd: <http://www.w3.org/2001/XMLSchema#> 
PREFIX rml: <http://semweb.mmlab.be/ns/rml#>

PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#> 
PREFIX bacdive: <http://kg.bacdive.dsmz.de/>

select distinct ?strain  ?pathogenicity_human ?ab_resistant 

where
{
    
    ?strain a bacdive:Strain;
         bacdive:has_risk_assessment ?risk;
         bacdive:has_met_antibiotica ?met;
         bacdive:has_culture_temp ?culture_temp;
         optional { ?strain bacdive:has_genus "Staphylococcus". }
       ?risk a bacdive:Risk_assessment;
        bacdive:has_pathogenicity_human ?pathogenicity_human.
    ?met a bacdive:Met_antibiotica;
       bacdive:has_ab_resistant ?ab_resistant.
     ?culture_temp a bacdive:Culture_temp;
       bacdive:has_test_type "optimum".

    }
       

Question 5) How are cultivation pH (rgEND < 6.00 + ability=positive) and the origin classification (is_link) related.

PREFIX owl: <http://www.w3.org/2002/07/owl#> 
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#> 
PREFIX xsd: <http://www.w3.org/2001/XMLSchema#> 
PREFIX rml: <http://semweb.mmlab.be/ns/rml#>
PREFIX ql: <http://semweb.mmlab.be/ns/ql#> 
PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#> 
PREFIX bacdive: <http://kg.bacdive.dsmz.de/>

select distinct ?strain ?rg1 ?origin ?cat1_id  ?cat2_id ?cat3_id ?cat1 ?cat2 ?cat3 ?origin_sample_type

    {
    
    ?strain a bacdive:Strain;
         bacdive:has_culture_ph ?culture_ph;
         bacdive:has_is_link ?link.
    
    ?culture_ph a bacdive:Culture_pH;
        bacdive:has_rgEnd ?rg;
        bacdive:has_ability "positive". 

 ?link a bacdive:Is_link.
    optional {
      ?link   bacdive:has_origin ?origin;
              bacdive:has_Cat1_link ?cat1_id.}
       
     ?cat1_id a bacdive:Is_cat1;
       bacdive:has_cat1 ?cat1.
    
 ?cat2_id a bacdive:Is_cat2;
       bacdive:has_cat2 ?cat2.
    
    
      
       ?cat3_id a bacdive:Is_cat3;
       bacdive:has_cat3 ?cat3.

       
     ?origin a bacdive:Origin;
            bacdive:has_sample_type ?origin_sample_type

    
       
    BIND(STRDT(STR(?rg), xsd:float) AS ?rg1)
    filter( ?rg1 < "6.00"^^xsd:float)
 }
